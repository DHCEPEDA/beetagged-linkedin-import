  <script>
    let processedContacts = [];
    let allTags = new Set();

    // File upload handling
    const uploadArea = document.getElementById('uploadArea');
    const fileInput = document.getElementById('fileInput');
    const status = document.getElementById('status');

    uploadArea.addEventListener('click', () => fileInput.click());

    uploadArea.addEventListener('dragover', (e) => {
      e.preventDefault();
      uploadArea.classList.add('dragover');
    });

    uploadArea.addEventListener('dragleave', () => {
      uploadArea.classList.remove('dragover');
    });

    uploadArea.addEventListener('drop', (e) => {
      e.preventDefault();
      uploadArea.classList.remove('dragover');
      const file = e.dataTransfer.files[0];
      if (file && file.name.endsWith('.csv')) {
        processFile(file);
      } else {
        showStatus('Please upload a CSV file', 'error');
      }
    });

    fileInput.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (file) {
        processFile(file);
      }
    });

    function showStatus(message, type) {
      status.textContent = message;
      status.className = `status ${type}`;
      status.style.display = 'block';
      
      if (type === 'success' || type === 'info') {
        setTimeout(() => {
          status.style.display = 'none';
        }, 3000);
      }
    }

    function processFile(file) {
      showStatus('Processing file...', 'info');
      
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const csv = e.target.result;
          const contacts = parseCSV(csv);
          
          if (contacts.length === 0) {
            showStatus('No contacts found in file', 'error');
            return;
          }

          processedContacts = contacts.map(contact => processContact(contact));
          displayResults();
          showStatus(`Successfully processed ${contacts.length} contacts!`, 'success');
        } catch (error) {
          showStatus(`Error processing file: ${error.message}`, 'error');
          console.error('Processing error:', error);
        }
      };
      
      reader.onerror = () => {
        showStatus('Error reading file', 'error');
      };
      
      reader.readAsText(file);
    }

    function parseCSV(csv) {
      const lines = csv.split('\n').filter(line => line.trim());
      if (lines.length < 2) return [];

      const headers = lines[0].split(',').map(h => h.trim().replace(/"/g, ''));
      const contacts = [];

      for (let i = 1; i < lines.length; i++) {
        const values = parseCSVLine(lines[i]);
        if (values.length === headers.length) {
          const contact = {};
          headers.forEach((header, index) => {
            contact[header] = values[index].trim().replace(/"/g, '');
          });
          contacts.push(contact);
        }
      }

      return contacts;
    }

    function parseCSVLine(line) {
      const values = [];
      let current = '';
      let inQuotes = false;

      for (let i = 0; i < line.length; i++) {
        const char = line[i];
        
        if (char === '"') {
          inQuotes = !inQuotes;
        } else if (char === ',' && !inQuotes) {
          values.push(current);
          current = '';
        } else {
          current += char;
        }
      }
      values.push(current);
      
      return values;
    }

    function processContact(contact) {
      // Handle different CSV formats (LinkedIn exports can vary)
      const firstName = contact['First Name'] || contact['first name'] || '';
      const lastName = contact['Last Name'] || contact['last name'] || '';
      const email = contact['Email Address'] || contact['email'] || contact['Email'] || '';
      const company = contact['Company'] || contact['company'] || '';
      const position = contact['Position'] || contact['position'] || contact['Job Title'] || '';
      const connectedOn = contact['Connected On'] || contact['connected on'] || '';

      const tags = generateTags(contact);
      tags.forEach(tag => allTags.add(tag));

      return {
        firstName,
        lastName,
        fullName: `${firstName} ${lastName}`.trim(),
        email,
        company,
        position,
        connectedOn,
        tags
      };
    }

    function generateTags(contact) {
      const tags = [];

      // Company tag
      const company = contact['Company'] || contact['company'] || '';
      if (company) {
        tags.push(`company:${company}`);
      }

      // Position/Title tag
      const position = contact['Position'] || contact['position'] || contact['Job Title'] || '';
      if (position) {
        // Extract key position terms
        const positionLower = position.toLowerCase();
        if (positionLower.includes('ceo') || positionLower.includes('founder')) {
          tags.push('role:founder');
        } else if (positionLower.includes('cto') || positionLower.includes('chief technology')) {
          tags.push('role:cto');
        } else if (positionLower.includes('engineer')) {
          tags.push('role:engineer');
        } else if (positionLower.includes('manager')) {
          tags.push('role:manager');
        } else if (positionLower.includes('director')) {
          tags.push('role:director');
        } else if (positionLower.includes('vp') || positionLower.includes('vice president')) {
          tags.push('role:vp');
        } else if (positionLower.includes('analyst')) {
          tags.push('role:analyst');
        } else if (positionLower.includes('designer')) {
          tags.push('role:designer');
        } else if (positionLower.includes('developer')) {
          tags.push('role:developer');
        }
        
        tags.push(`position:${position}`);
      }

      // Connection year
      const connectedOn = contact['Connected On'] || contact['connected on'] || '';
      if (connectedOn) {
        const year = connectedOn.split(/[-\/\s]/)[0];
        if (year && year.length === 4) {
          tags.push(`connected:${year}`);
        }
      }

      // LinkedIn tag (default)
      tags.push('source:linkedin');

      return tags;
    }

    function displayResults() {
      document.getElementById('uploadSection').style.display = 'none';
      document.getElementById('results').style.display = 'block';

      // Update stats
      document.getElementById('totalContacts').textContent = processedContacts.length;
      document.getElementById('totalTags').textContent = allTags.size;
      
      const companies = new Set(processedContacts.map(c => c.company).filter(c => c));
      document.getElementById('totalCompanies').textContent = companies.size;

      // Display first 10 contacts
      const contactList = document.getElementById('contactList');
      contactList.innerHTML = '';
      
      processedContacts.slice(0, 10).forEach(contact => {
        const item = document.createElement('div');
        item.className = 'contact-item';
        
        const details = [];
        if (contact.company) details.push(contact.company);
        if (contact.position) details.push(contact.position);
        if (contact.email) details.push(contact.email);
        
        item.innerHTML = `
          <div class="contact-name">${contact.fullName || 'No Name'}</div>
          <div class="contact-details">${details.join(' ‚Ä¢ ')}</div>
          <div class="tags">
            ${contact.tags.map(tag => `<span class="tag">${tag}</span>`).join('')}
          </div>
        `;
        
        contactList.appendChild(item);
      });
    }

    function downloadJSON() {
      const data = {
        exportDate: new Date().toISOString(),
        totalContacts: processedContacts.length,
        contacts: processedContacts
      };
      
      const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
      downloadBlob(blob, 'beetagged-contacts.json');
    }

    function downloadCSV() {
      const headers = ['First Name', 'Last Name', 'Email', 'Company', 'Position', 'Tags', 'Connected On'];
      const rows = processedContacts.map(contact => [
        contact.firstName,
        contact.lastName,
        contact.email,
        contact.company,
        contact.position,
        contact.tags.join('; '),
        contact.connectedOn
      ]);
      
      const csv = [
        headers.join(','),
        ...rows.map(row => row.map(cell => `"${cell}"`).join(','))
      ].join('\n');
      
      const blob = new Blob([csv], { type: 'text/csv' });
      downloadBlob(blob, 'beetagged-contacts.csv');
    }

    function downloadBlob(blob, filename) {
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = filename;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      URL.revokeObjectURL(url);
    }

    function generateAccessCode() {
      // Generate a 6-digit access code
      const code = Math.floor(100000 + Math.random() * 900000).toString();
      document.getElementById('accessCode').textContent = code;
      document.getElementById('accessCodeSection').style.display = 'block';
      
      // In a real implementation, this would send data to a server
      // For now, we'll store it in localStorage with the code as key
      const data = {
        code: code,
        timestamp: new Date().toISOString(),
        contacts: processedContacts,
        expiresAt: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString() // 24 hours
      };
      
      localStorage.setItem(`beetagged_${code}`, JSON.stringify(data));
      
      showStatus('Access code generated! Valid for 24 hours.', 'success');
    }

    function copyAccessCode() {
      const code = document.getElementById('accessCode').textContent;
      navigator.clipboard.writeText(code).then(() => {
        showStatus('Access code copied to clipboard!', 'success');
      }).catch(() => {
        showStatus('Failed to copy code', 'error');
      });
    }

    function reset() {
      document.getElementById('uploadSection').style.display = 'block';
      document.getElementById('results').style.display = 'none';
      document.getElementById('accessCodeSection').style.display = 'none';
      document.getElementById('fileInput').value = '';
      processedContacts = [];
      allTags.clear();
    }
  </script>
